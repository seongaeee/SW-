### 솔루션
```
처음 문제를 보았을때 처음 보는 배열의 형태 그림이 엄청 많이 있어서 포기했다가 다시 풀어보았다.
어떻게 구현을 할까 어려워하다가 다른 사람들의 블로그를 참고하였다.
두개의 배열로 나누어, 똑같은 방식으로 두번 처리하면 되기 때문에 생각보다 쉬운문제였다.
복잡한 그림이 나올때는 분해하여 고려해보자!
```

### 나의 코드
```
package SW역량테스트;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Iterator;
import java.util.StringTokenizer;

public class bj_20061_모노미노도미노2 {
	
	static int N, block[][], green[][], blue[][], totalScore;
	static BufferedReader br=new BufferedReader(new InputStreamReader(System.in));
	static StringTokenizer st;
	
	public static void main(String[] args) throws NumberFormatException, IOException {
		
		N=Integer.parseInt(br.readLine());
		block=new int[N][3];
		green=new int[7][4]; //행 0번째 줄은 내리는 작업을 위해 추가
		blue=new int[7][4];
		
		for (int i = 0; i < N; i++) {
			st=new StringTokenizer(br.readLine());
			int t=Integer.parseInt(st.nextToken());
			int r=Integer.parseInt(st.nextToken());
			int c=Integer.parseInt(st.nextToken());
			block[i]=new int[] {t,r,c};
		}
		
		process();
		
		int cnt=0;
		for (int i = 1; i < 7; i++) {
			for (int j = 0; j < 4; j++) {
				if(green[i][j]!=0) cnt++;
				if(blue[i][j]!=0) cnt++;
			}
		}
		System.out.println(totalScore);
		System.out.println(cnt);
	}

	private static void process() {
		
		for (int n = 0; n < N; n++) {
			
			int t=block[n][0];
			int r=block[n][1];
			int c=block[n][2];
			int tBlue=1;
			if(t==2) tBlue=3;
			else if(t==3) tBlue=2;
			
			//1.블록 내리기
			down(t,c,green);
			down(tBlue,r,blue);
			
			//2.블록으로 찼는지 확인 & 삭제
			checkFull(green);
			checkFull(blue);
			
			//3.특별한 칸 확인 & 삭제
			checkSpecial(green);
			checkSpecial(blue);
			
		}
		
	}

	private static void checkSpecial(int[][] board) {
		
		for (int i = 1; i <= 2; i++) { //특별한 칸의 행
			int cnt=0; //한 행에서 0이 아닌 원소 개수
			for (int j = 0; j < 4; j++) {
				if(board[i][j]!=0) cnt++;
			}
			if(cnt!=0) 
				remove(6,board);
		}
	}

	private static void checkFull(int[][] board) {
		
		for (int i = 0; i < 7; i++) {
			int cnt=0; //한 행에서 0이 아닌 원소 개수
			for (int j = 0; j < 4; j++) {
				if(board[i][j]!=0) cnt++;
			}
			if(cnt==4) {
				remove(i,board);
				totalScore++;
			}
		}
	}

	private static void remove(int removeR, int[][] board) {
		
		for (int r = removeR-1; r >= 0; r--) {
			for (int c = 0; c < 4; c++) {
				board[r+1][c]=board[r][c];
			}
		}
	}

	private static void down(int t, int c, int[][] board) {
		
		if(t==1) {
			int r=1; //넣을 열 위치
			while(r<7 && board[r][c]==0) 
				r++;
			board[--r][c]=t;
		}
		else if(t==2) {
			int r=1; 
			while(r<7 && board[r][c]==0) r++;
			int r2=1; 
			while(r2<7 && board[r2][c+1]==0) r2++;
			
			r=Math.min(r, r2);
			board[--r][c]=t;
			board[r][++c]=t;
		}
		else if(t==3) {
			int r=1;
			while(r<7 && board[r][c]==0) r++;
			board[--r][c]=t;
			board[--r][c]=t;
		}
	}
}
```
